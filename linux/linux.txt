#Найдите в файле access.log все упоминания строки "404"
    cd ../logs/ #Перехожу в папку logs
    grep -с "access.log" 404
#На основании файла 404.html сделайте файл 403.html заменив в его содержимом все упоминания 404 на 403
    cat 404.html > 403.html#Копия содержимого файла 404.html
    sed -i 's/404'/403/g/ 403.html#Поиск и замена 404 на 403

#Модифицируйте права доступа и владельца файла index.html следующим образом:
Текущий пользователь должен иметь права на чтение/запись 
Группа www-data на чтение Все остальные не имеют доступа к файлу
    sudo groupadd www-data #Создание группы
    chmod g+r index.html #Предоставляю группе права доступа на чтение
    chmod o-rwx index.html #Ограничиваю доступ к файлу

#Сделайте символическую ссылку на файл access.log в папке html
    cd ../logs/ #Выхожу из текущей деректории var и перехожу  в logs
    ln -s access.log softlink #Создаю символическую ссылку

#Выведите на экран все файлы в папке var
